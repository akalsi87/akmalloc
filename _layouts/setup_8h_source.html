60 static void* ak_os_alloc(size_t sz)
61 {
62  static const ak_sz pgsz = AKMALLOC_DEFAULT_PAGE_SIZE;
63  (void)(pgsz);
64  AKMALLOC_ASSERT_ALWAYS(pgsz == ak_page_size());
65  void* mem = AKMALLOC_MMAP(sz);
66  DBG_PRINTF("osmap,%p,%zu,%zu pages,iswhole %d\n", mem, sz, sz/AKMALLOC_DEFAULT_PAGE_SIZE, sz == AKMALLOC_DEFAULT_PAGE_SIZE*(sz/AKMALLOC_DEFAULT_PAGE_SIZE));
67  return mem;
68 }
