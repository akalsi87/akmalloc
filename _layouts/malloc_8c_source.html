56 #define ak_ensure_malloc_state_init() \
57 { \
58  if (ak_unlikely(!MALLOC_INIT)) { \
59  AKMALLOC_LOCK_ACQUIRE(ak_as_ptr(MALLOC_INIT_LOCK)); \
60  if (MALLOC_INIT != 1) { \
61  GMSTATE = &MALLOC_ROOT; \
62  ak_malloc_init_state(GMSTATE); \
63  MALLOC_INIT = 1; \
64  } \
65  AKMALLOC_LOCK_RELEASE(ak_as_ptr(MALLOC_INIT_LOCK)); \
66  } \
67  AKMALLOC_ASSERT(MALLOC_ROOT.init); \
68 }
