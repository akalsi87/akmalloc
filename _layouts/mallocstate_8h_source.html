95 // Coalescing allocs give 16-byte aligned memory where the preamble
96 // uses three bits. The fourth bit is always free. We use that bit
97 // to distinguish slabs from coalesced outputs, and mmap-outputs.
98 //
99 // xxx0 - coalesce
100 // 0101 - slab
101 // 1001 - mmap
102 #define ak_alloc_type_bits(p) \
103  ((*(((const ak_sz*)(p)) - 1)) & (AK_COALESCE_ALIGN - 1))
104 
105 #define ak_alloc_type_coalesce(sz) \
106  ((((ak_sz)sz) & ((ak_sz)8)) == 0)
107 
108 #define ak_alloc_type_slab(sz) \
109  ((((ak_sz)sz) & (AK_COALESCE_ALIGN - 1)) == 10)
110 
111 #define ak_alloc_type_mmap(sz) \
112  ((((ak_sz)sz) & (AK_COALESCE_ALIGN - 1)) == 9)
113 
114 #define ak_alloc_mark_coalesce(p) ((void)(p))
115 
116 #define ak_alloc_mark_slab(p) \
117  *(((ak_sz*)(p)) - 1) = ((ak_sz)10)
118 
119 #define ak_alloc_mark_mmap(p) \
120  *(((ak_sz*)(p)) - 1) = ((ak_sz)9)

194 static void ak_try_reclaim_memory(ak_malloc_state* m)
195 {
196  // for each slab, reclaim empty pages
197  for (ak_sz i = 0; i < NSLABS; ++i) {
198  ak_slab_root* s = ak_as_ptr(m->slabs[i]);
199  ak_slab_release_pages(s, ak_as_ptr(s->empty_root), AK_U32_MAX);
200  s->nempty = 0;
201  s->release = 0;
202  }
203  // return unused segments in ca
204  for (ak_sz i = 0; i < NCAROOTS; ++i) {
205  ak_ca_root* ca = ak_as_ptr(m->ca[i]);
206  ak_ca_return_os_mem(ak_as_ptr(ca->empty_root), AK_U32_MAX);
207  ca->nempty = 0;
208  ca->release = 0;
209  }
210 
211  // all memory in mmap-ed regions is being used. we return pages immediately
212  // when they are free'd.
213 }

266 ak_inline static void* ak_try_alloc(ak_malloc_state* m, size_t sz)
267 {
268  void* retmem = AK_NULLPTR;
269  ak_sz modsz = ak_slab_mod_sz(sz);
270  if (modsz <= MIN_SMALL_REQUEST) {
271  retmem = ak_try_slab_alloc(m, modsz);
272  DBG_PRINTF("a,slab,%p,%llu\n", retmem, modsz);
273  } else if (sz < MMAP_SIZE) {
274  const ak_sz alnsz = ak_ca_aligned_size(sz);
275  ak_ca_root* proot = ak_find_ca_root(m, sz);
276  retmem = ak_try_coalesce_alloc(m, proot, alnsz);
277  DBG_PRINTF("a,ca[%d],%p,%llu\n", (int)(proot-ak_as_ptr(m->ca[0])), retmem, alnsz);
278  } else {
279  sz += sizeof(ak_ca_segment);
280  const ak_sz actsz = ak_ca_aligned_segment_size(sz);
281  retmem = ak_try_alloc_mmap(m, actsz);
282  DBG_PRINTF("a,mmap,%p,%llu\n", retmem, actsz);
283  }
284 
285  return retmem;
286 }

289 static void* ak_aligned_alloc_from_state_no_checks(ak_malloc_state* m, size_t aln, size_t sz)
290 {
291  ak_sz req = ak_ca_aligned_size(sz);
292  req += aln + (2 * sizeof(ak_alloc_node)) + (2 * sizeof(ak_free_list_node));
293  char* mem = AK_NULLPTR;
294  // must request from coalesce alloc so we can return the extra piece
295  ak_ca_root* ca = ak_find_ca_root(m, req);
296 
297  mem = (char*)ak_ca_alloc(ca, req);
298  ak_alloc_node* node = ak_ptr_cast(ak_alloc_node, mem) - 1;
299  if (ak_likely(mem)) {
300  if ((((ak_sz)mem) & (aln - 1)) != 0) {
301  // misaligned
302  AK_CA_LOCK_ACQUIRE(ca);
303  char* alnpos = (char*)(((ak_sz)(mem + aln - 1)) & ~(aln - 1));
304  ak_alloc_node* alnnode = ak_ptr_cast(ak_alloc_node, alnpos) - 1;
305  AKMALLOC_ASSERT((ak_sz)(alnpos - mem) >= (sizeof(ak_free_list_node) + sizeof(ak_alloc_node)));
306  ak_sz actsz = ak_ca_to_sz(node->currinfo);
307  int islast = ak_ca_is_last(node);
308 
309  ak_ca_set_sz(ak_as_ptr(node->currinfo), alnpos - mem - sizeof(ak_alloc_node));
310  ak_ca_set_is_last(ak_as_ptr(node->currinfo), 0);
311  ak_ca_set_is_free(ak_as_ptr(node->currinfo), 1);
312 
313  ak_ca_set_sz(ak_as_ptr(alnnode->currinfo), actsz - ak_ca_to_sz(node->currinfo));
314  ak_ca_set_is_last(ak_as_ptr(alnnode->currinfo), islast);
315  ak_ca_set_is_free(ak_as_ptr(alnnode->currinfo), 0);
316 
317  ak_ca_update_footer(node);
318  ak_ca_update_footer(alnnode);
319  mem = alnpos;
320  AK_CA_LOCK_RELEASE(ca);
321  }
322  return mem;
323  }
324  return AK_NULLPTR;
325 }

400 ak_inline static void ak_free_to_state(ak_malloc_state* m, void* mem)
401 {
402  if (ak_likely(mem)) {

408  DBG_PRINTF("d,slab,%p,%llu\n", mem, ussize);

414  ak_ca_segment_unlink(seg);

416  AKMALLOC_LOCK_RELEASE(ak_as_ptr(m->MAP_LOCK));

496 {

525 static void* ak_aligned_alloc_from_state(ak_malloc_state* m, size_t aln, size_t sz)
526 {
527  if (aln <= AK_COALESCE_ALIGN) {
528  return ak_malloc_from_state(m, sz);
529  }
530  if ((aln & AK_SZ_ONE) || (aln & (aln - 1))) {
531  size_t a = AK_COALESCE_ALIGN << 1;
532  while (a < aln) {
533  a = (a << 1);
534  }
535  aln = a;
536  }
537  return ak_aligned_alloc_from_state_no_checks(m, aln, sz);
